52  static void composeCompound(Statement taskContent, Statement beliefContent,
53  int index, Memory memory) {
54  if ((!memory.getCurrentTask().getSentence().isJudgment())
55  || (taskContent.getClass() != beliefContent.getClass())) {
56  return;
57  }
58  Term componentT = taskContent.componentAt(1 - index);
59  Term componentB = beliefContent.componentAt(1 - index);
60  Term componentCommon = taskContent.componentAt(index);
61  if ((componentT instanceof CompoundTerm)
62  && ((CompoundTerm) componentT).containAllComponents(componentB)) {
63  decomposeCompound((CompoundTerm) componentT, componentB,
64  componentCommon, index, true, memory);
65  return;
66  } else if ((componentB instanceof CompoundTerm)
67  && ((CompoundTerm) componentB).containAllComponents(componentT)) {
68  decomposeCompound((CompoundTerm) componentB, componentT,
69  componentCommon, index, false, memory);
70  return;
71  }
72  TruthValue truthT = memory.getCurrentTask().getSentence().getTruth();
73  TruthValue truthB = memory.getCurrentBelief().getTruth();
74  TruthValue truthOr = TruthFunctions.union(truthT, truthB);
75  TruthValue truthAnd = TruthFunctions.intersection(truthT, truthB);
76  Term termOr = null;
77  Term termAnd = null;
78  if (index == 0) {
79  if (taskContent instanceof Inheritance) {
80  termOr = IntersectionInt.make(componentT, componentB, memory);
81  if (truthB.isNegative()) {
82  if (!truthT.isNegative()) {
83  termAnd = DifferenceExt.make(componentT, componentB,
84  memory);
85  truthAnd = TruthFunctions.intersection(truthT,
86  TruthFunctions.negation(truthB));
87  }
88  } else if (truthT.isNegative()) {
89  termAnd = DifferenceExt
90  .make(componentB, componentT, memory);
91  truthAnd = TruthFunctions.intersection(truthB,
92  TruthFunctions.negation(truthT));
93  } else {
94  termAnd = IntersectionExt.make(componentT, componentB,
95  memory);
96  }
97  } else if (taskContent instanceof Implication) {
98  termOr = Disjunction.make(componentT, componentB, memory);
99  termAnd = Conjunction.make(componentT, componentB, memory);
100  }
101  processComposed(taskContent, componentCommon, termOr, truthOr,
102  memory);
103  processComposed(taskContent, componentCommon, termAnd, truthAnd,
104  memory);
105  } else { // index == 1
106  if (taskContent instanceof Inheritance) {
107  termOr = IntersectionExt.make(componentT, componentB, memory);
108  if (truthB.isNegative()) {
109  if (!truthT.isNegative()) {
110  termAnd = DifferenceInt.make(componentT, componentB,
111  memory);
112  truthAnd = TruthFunctions.intersection(truthT,
113  TruthFunctions.negation(truthB));
114  }
115  } else if (truthT.isNegative()) {
116  termAnd = DifferenceInt
117  .make(componentB, componentT, memory);
118  truthAnd = TruthFunctions.intersection(truthB,
119  TruthFunctions.negation(truthT));
120  } else {
121  termAnd = IntersectionInt.make(componentT, componentB,
122  memory);
123  }
124  } else if (taskContent instanceof Implication) {
125  termOr = Conjunction.make(componentT, componentB, memory);
126  termAnd = Disjunction.make(componentT, componentB, memory);
127  }
128  processComposed(taskContent, termOr, componentCommon, truthOr,
129  memory);
130  processComposed(taskContent, termAnd, componentCommon, truthAnd,
131  memory);
132  }
133  if (taskContent instanceof Inheritance) {
134  introVarOuter(taskContent, beliefContent, index, memory);// introVarImage(taskContent,
135  // beliefContent,
136  // index,
137  // memory);
138  }

189  private static void decomposeCompound(CompoundTerm compound,
190  Term component, Term term1, int index, boolean compoundTask,
191  Memory memory) {
192  if (compound instanceof Statement) {
193  return;
194  }
195  Term term2 = CompoundTerm.reduceComponents(compound, component, memory);
196  if (term2 == null) {
197  return;
198  }
199  Task task = memory.getCurrentTask();
200  Sentence sentence = task.getSentence();
201  Sentence belief = memory.getCurrentBelief();
202  Statement oldContent = (Statement) task.getContent();
203  TruthValue v1, v2;
204  if (compoundTask) {
205  v1 = sentence.getTruth();
206  v2 = belief.getTruth();
207  } else {
208  v1 = belief.getTruth();
209  v2 = sentence.getTruth();
210  }
211  TruthValue truth = null;
212  Term content;
213  if (index == 0) {
214  content = Statement.make(oldContent, term1, term2, memory);
215  if (content == null) {
216  return;
217  }
218  if (oldContent instanceof Inheritance) {
219  if (compound instanceof IntersectionExt) {
220  truth = TruthFunctions.reduceConjunction(v1, v2);
221  } else if (compound instanceof IntersectionInt) {
222  truth = TruthFunctions.reduceDisjunction(v1, v2);
223  } else if ((compound instanceof SetInt)
224  && (component instanceof SetInt)) {
225  truth = TruthFunctions.reduceConjunction(v1, v2);
226  } else if ((compound instanceof SetExt)
227  && (component instanceof SetExt)) {
228  truth = TruthFunctions.reduceDisjunction(v1, v2);
229  } else if (compound instanceof DifferenceExt) {
230  if (compound.componentAt(0).equals(component)) {
231  truth = TruthFunctions.reduceDisjunction(v2, v1);
232  } else {
233  truth = TruthFunctions.reduceConjunctionNeg(v1, v2);
234  }
235  }
236  } else if (oldContent instanceof Implication) {
237  if (compound instanceof Conjunction) {
238  truth = TruthFunctions.reduceConjunction(v1, v2);
239  } else if (compound instanceof Disjunction) {
240  truth = TruthFunctions.reduceDisjunction(v1, v2);
241  }
242  }
243  } else {
244  content = Statement.make(oldContent, term2, term1, memory);
245  if (content == null) {
246  return;
247  }
248  if (oldContent instanceof Inheritance) {
249  if (compound instanceof IntersectionInt) {
250  truth = TruthFunctions.reduceConjunction(v1, v2);
251  } else if (compound instanceof IntersectionExt) {
252  truth = TruthFunctions.reduceDisjunction(v1, v2);
253  } else if ((compound instanceof SetExt)
254  && (component instanceof SetExt)) {
255  truth = TruthFunctions.reduceConjunction(v1, v2);
256  } else if ((compound instanceof SetInt)
257  && (component instanceof SetInt)) {
258  truth = TruthFunctions.reduceDisjunction(v1, v2);
259  } else if (compound instanceof DifferenceInt) {
260  if (compound.componentAt(1).equals(component)) {
261  truth = TruthFunctions.reduceDisjunction(v2, v1);
262  } else {
263  truth = TruthFunctions.reduceConjunctionNeg(v1, v2);
264  }
265  }
266  } else if (oldContent instanceof Implication) {
267  if (compound instanceof Disjunction) {
268  truth = TruthFunctions.reduceConjunction(v1, v2);
269  } else if (compound instanceof Conjunction) {
270  truth = TruthFunctions.reduceDisjunction(v1, v2);
271  }
272  }
273  }
274  if (truth != null) {
275  BudgetValue budget = BudgetFunctions.compoundForward(truth,
276  content, memory);
277  memory.doublePremiseTask(content, truth, budget);
278  }

293  static void decomposeStatement(CompoundTerm compound, Term component,
294  boolean compoundTask, Memory memory) {
295  Task task = memory.getCurrentTask();
296  Sentence sentence = task.getSentence();
297  if (sentence.isQuestion()) {
298  return;
299  }
300  Sentence belief = memory.getCurrentBelief();
301  Term content = CompoundTerm.reduceComponents(compound, component,
302  memory);
303  if (content == null) {
304  return;
305  }
306  TruthValue v1, v2;
307  if (compoundTask) {
308  v1 = sentence.getTruth();
309  v2 = belief.getTruth();
310  } else {
311  v1 = belief.getTruth();
312  v2 = sentence.getTruth();
313  }
314  TruthValue truth = null;

322  }
323  } else {

345  private static void introVarOuter(Statement taskContent, Statement beliefContent, int index, Memory memory) {
346  TruthValue truthT = memory.getCurrentTask().getSentence().getTruth();
347  TruthValue truthB = memory.getCurrentBelief().getTruth();
348  Variable varInd = new Variable("$varInd1");
349  Variable varInd2 = new Variable("$varInd2");

351  HashMap<Term, Term> subs = new HashMap<Term, Term>();
352  if (index == 0) {
353  term11 = varInd;
354  term21 = varInd;
355  term12 = taskContent.getPredicate();
356  term22 = beliefContent.getPredicate();
357  if ((term12 instanceof ImageExt) && (term22 instanceof ImageExt)) {
358  commonTerm = ((ImageExt) term12).getTheOtherComponent();
359  if ((commonTerm == null) || !((ImageExt) term22).containTerm(commonTerm)) {
360  commonTerm = ((ImageExt) term22).getTheOtherComponent();
361  if ((commonTerm == null) || !((ImageExt) term12).containTerm(commonTerm)) {
362  commonTerm = null;
363  }
364  }
365  if (commonTerm != null) {
366  subs.put(commonTerm, varInd2);
367  ((ImageExt) term12).applySubstitute(subs);
368  ((ImageExt) term22).applySubstitute(subs);
369  }
370  }
371  } else {
372  term11 = taskContent.getSubject();
373  term21 = beliefContent.getSubject();
374  term12 = varInd;
375  term22 = varInd;
376  if ((term11 instanceof ImageInt) && (term21 instanceof ImageInt)) {
377  commonTerm = ((ImageInt) term11).getTheOtherComponent();
378  if ((commonTerm == null) || !((ImageInt) term21).containTerm(commonTerm)) {
379  commonTerm = ((ImageInt) term21).getTheOtherComponent();
380  if ((commonTerm == null) || !((ImageExt) term11).containTerm(commonTerm)) {
381  commonTerm = null;
382  }
383  }
384  if (commonTerm != null) {
385  subs.put(commonTerm, varInd2);
386  ((ImageInt) term11).applySubstitute(subs);
387  ((ImageInt) term21).applySubstitute(subs);
388  }
389  }
390  }
391  Statement state1 = Inheritance.make(term11, term12, memory);
392  Statement state2 = Inheritance.make(term21, term22, memory);
393  Term content = Implication.make(state1, state2, memory);
394  TruthValue truth = TruthFunctions.induction(truthT, truthB);
395  BudgetValue budget = BudgetFunctions.compoundForward(truth, content, memory);
396  memory.doublePremiseTask(content, truth, budget);
397  content = Implication.make(state2, state1, memory);
398  truth = TruthFunctions.induction(truthB, truthT);
399  budget = BudgetFunctions.compoundForward(truth, content, memory);
400  memory.doublePremiseTask(content, truth, budget);
401  content = Equivalence.make(state1, state2, memory);
402  truth = TruthFunctions.comparison(truthT, truthB);
403  budget = BudgetFunctions.compoundForward(truth, content, memory);
404  memory.doublePremiseTask(content, truth, budget);
405  Variable varDep = new Variable("#varDep");
406  if (index == 0) {
407  state1 = Inheritance.make(varDep, taskContent.getPredicate(), memory);
408  state2 = Inheritance.make(varDep, beliefContent.getPredicate(), memory);
409  } else {
410  state1 = Inheritance.make(taskContent.getSubject(), varDep, memory);
411  state2 = Inheritance.make(beliefContent.getSubject(), varDep, memory);
412  }
413  content = Conjunction.make(state1, state2, memory);
414  truth = TruthFunctions.intersection(truthT, truthB);
415  budget = BudgetFunctions.compoundForward(truth, content, memory);
416  memory.doublePremiseTask(content, truth, budget, false);
417  }
