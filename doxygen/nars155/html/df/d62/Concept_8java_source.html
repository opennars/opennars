114  private void processJudgment(Task task) {
115  Sentence judg = task.getSentence();
116  Sentence oldBelief = evaluation(judg, beliefs);
117  if (oldBelief != null) {
118  Stamp newStamp = judg.getStamp(), oldStamp = oldBelief.getStamp();
119  if (newStamp.equals(oldStamp)) {
120  task.getBudget().decPriority(0); // duplicated task
121  return;
122  } else if (LocalRules.revisible(judg, oldBelief)) {
123  memory.setNewStamp(Stamp.make(newStamp, oldStamp, memory.getTime()));
124  if (memory.getNewStamp() != null) {
125  memory.setCurrentBelief(oldBelief);
126  LocalRules.revision(judg, oldBelief, false, memory);
127  }
128  }
129  }
130  if (task.getBudget().aboveThreshold()) {
131  for (Task ques : questions) {
132 // LocalRules.trySolution(ques.getSentence(), judg, ques, memory);
133  LocalRules.trySolution(judg, ques, memory);
134  }
135  addToTable(judg, beliefs, Memory.Parameters.MAXIMUM_BELIEF_LENGTH);
136  }
137  }

144  public float processQuestion(Task task) {
145  Sentence ques = task.getSentence();
146  boolean newQuestion = true;
147  if (questions != null) {
148  for (Task t : questions) {
149  Sentence q = t.getSentence();
150  if (q.getContent().equals(ques.getContent())) {
151  ques = q;
152  newQuestion = false;
153  break;
154  }
155  }
156  }
157  if (newQuestion) {
158  questions.add(task);
159  }
160  if (questions.size() > Memory.Parameters.MAXIMUM_QUESTIONS_LENGTH) {
161  questions.remove(0); // FIFO
162  }
163  Sentence newAnswer = evaluation(ques, beliefs);
164  if (newAnswer != null) {
165 // LocalRules.trySolution(ques, newAnswer, task, memory);
166  LocalRules.trySolution(newAnswer, task, memory);
167  return newAnswer.getTruth().getExpectation();
168  } else {
169  return 0.5f;
170  }
171  }

214  private void addToTable(Sentence newSentence, ArrayList<Sentence> table, int capacity) {
215  float rank1 = BudgetFunctions.rankBelief(newSentence); // for the new isBelief
216  Sentence judgment2;
217  float rank2;
218  int i;
219  for (i = 0; i < table.size(); i++) {
220  judgment2 = (Sentence) table.get(i);
221  rank2 = BudgetFunctions.rankBelief(judgment2);
222  if (rank1 >= rank2) {
223  if (newSentence.equivalentTo(judgment2)) {
224  return;
225  }
226  table.add(i, newSentence);
227  break;
228  }
229  }
230  if (table.size() >= capacity) {
231  while (table.size() > capacity) {
232  table.remove(table.size() - 1);
233  }
234  } else if (i == table.size()) {
235  table.add(newSentence);
236  }
237  }

245  private Sentence evaluation(Sentence query, Iterable<Sentence> list) {
246  if (list == null) {
247  return null;
248  }
249  float currentBest = 0;
250  float beliefQuality;
251  Sentence candidate = null;
252  for (Sentence judg : list) {
253  beliefQuality = LocalRules.solutionQuality(query, judg);
254  if (beliefQuality > currentBest) {
255  currentBest = beliefQuality;
256  candidate = judg;
257  }
258  }
259  return candidate;
260  }

281  public void buildTermLinks(BudgetStruct taskBudget) {
282  Term t;
283  Concept concept;
284  TermLink termLink1, termLink2;
285  if (termLinkTemplates.size() > 0) {
286  BudgetValue subBudget = BudgetFunctions.distributeAmongLinks(taskBudget, termLinkTemplates.size());
287  if (subBudget.aboveThreshold()) {
288  for (TermLink template : termLinkTemplates) {
289  if (template.getType() != TermLink.TRANSFORM) {
290  t = template.getTarget();
291  concept = memory.getConcept(t);
292  if (concept != null) {
293  termLink1 = new TermLink(t, template, subBudget);
294  insertTermLink(termLink1); // this termLink to that
295  termLink2 = new TermLink(term, template, subBudget);
296  concept.insertTermLink(termLink2); // that termLink to this
297  if (t instanceof CompoundTerm) {
298  concept.buildTermLinks(subBudget);
299  }
300  }
301  }
302  }
303  }

400  /* ---------- main loop ---------- */
404  public void fire() {
405  TaskLink currentTaskLink = taskLinks.takeOut();
406  if (currentTaskLink == null) {
407  return;
408  }
409  memory.setCurrentTaskLink(currentTaskLink);
410  memory.setCurrentBeliefLink(null);
411  memory.getRecorder().append(" * Selected TaskLink: " + currentTaskLink + "\n");
412  Task task = currentTaskLink.getTargetTask();
413  memory.setCurrentTask(task); // one of the two places where this variable is set
414  if (currentTaskLink.getType() == TermLink.TRANSFORM) {
415  RuleTables.transformTask(currentTaskLink, memory); // to turn this into structural inference as below?
416  }
417  int termLinkCount = Memory.Parameters.MAX_REASONED_TERM_LINK;
418  while (memory.noResult() && (termLinkCount > 0)) {
419  TermLink termLink = termLinks.takeOut(currentTaskLink, memory.getTime());
420  if (termLink != null) {
421  memory.getRecorder().append(" * Selected TermLink: " + termLink + "\n");
422  memory.setCurrentBeliefLink(termLink);
423  RuleTables.reason(currentTaskLink, termLink, memory);
424  termLinks.putBack(termLink);
425  termLinkCount--;
426  } else {
427  termLinkCount = 0;
428  }
429  }
430  taskLinks.putBack(currentTaskLink);
431  }
