64  static void structuralCompose2(CompoundTerm compound, short index, Statement statement, short side, Memory memory) {
65  if (compound.equals(statement.componentAt(side))) {
66  return;
67  }
68  Term sub = statement.getSubject();
69  Term pred = statement.getPredicate();
70  ArrayList<Term> components = compound.cloneComponents();
71  if ((side == 0 && components.contains(pred)) || (side == 1 && components.contains(sub))) {
72  return;
73  }
74  if (side == 0) {
75  if (!components.contains(sub)) {
76  if (components.contains(pred)) {
77  if (sub instanceof CompoundTerm) {
78  return;
79  } else {
80  components.set(index, sub);
81  sub = CompoundTerm.make(compound, components, memory);
82  pred = compound;
83  }
84  }
85  else {
86  if (pred instanceof CompoundTerm) {
87  return;
88  } else {
89  sub = compound;
90  components.set(index, pred);
91  pred = CompoundTerm.make(compound, components, memory);
92  }
93  }
94  }
95  }
96  if (sub != null && pred != null) {
97  Term content = switchOrder(compound, index) ? Statement.make(statement, pred, sub, memory) : Statement.make(statement, sub, pred, memory);
98  if (content != null) {
99  Task task = memory.getCurrentTask();
100  Sentence sentence = task.getSentence();
101  TruthValue truth = sentence.getTruth();
102  BudgetValue budget;
103  if (sentence.isQuestion()) {
104  budget = BudgetFunctions.compoundBackwardWeak(content, memory);
105  } else {
106  if (compound.size() > 1) {
107  if (sentence.isJudgment()) {
108  truth = TruthFunctions.deduction(truth, RELIANCE);
109  } else {
110  return;
111  }
112  }
113  budget = compoundForward(truth, content, memory);
114  }
115  memory.singlePremiseTask(content, truth, budget);
116  }
117  }

127  static void structuralDecompose2(Statement statement, Memory memory) {
128  Term subj = statement.getSubject();
129  Term pred = statement.getPredicate();
130  if (subj.getClass() != pred.getClass()) {
131  return;
132  }
133  CompoundTerm sub = (CompoundTerm) subj;
134  CompoundTerm pre = (CompoundTerm) pred;
135  if (sub.size() != pre.size()) {
136  return;
137  }
138  int index = -1;
139  Term t1, t2;
140  for (int i = 0; i < sub.size(); i++) {
141  t1 = sub.componentAt(i);
142  t2 = pre.componentAt(i);
143  if (!t1.equals(t2)) {
144  if (index < 0) {
145  index = i;
146  } else {
147  return;
148  }
149  }
150  }
151  t1 = sub.componentAt(index);
152  t2 = pre.componentAt(index);
153  Term content;
154  if (switchOrder(sub, (short) index)) {
155  content = Statement.make(statement, t2, t1, memory);
156  } else {
157  content = Statement.make(statement, t1, t2, memory);
158  }
159  if (content == null) {
160  return;
161  }
162  Task task = memory.getCurrentTask();
163  Sentence sentence = task.getSentence();
164  TruthValue truth = sentence.getTruth();
165  BudgetValue budget;
166  if (sentence.isQuestion()) {
167  budget = compoundBackward(content, memory);
168  } else {
169  if (sub.size() > 1 && sentence.isJudgment()) {
170  return;
171 
172  }
173  budget = compoundForward(truth, content, memory);
174  }
175  memory.singlePremiseTask(content, truth, budget);

357  static void transformProductImage(Inheritance inh, CompoundTerm oldContent, short[] indices, Memory memory) {
358  Term subject = inh.getSubject();
359  Term predicate = inh.getPredicate();
360  if (inh.equals(oldContent)) {
361  if (subject instanceof CompoundTerm) transformSubjectPI((CompoundTerm) subject, predicate, memory);
362  if (predicate instanceof CompoundTerm) transformPredicatePI(subject, (CompoundTerm) predicate, memory);
363  } else {
364  short index = indices[indices.length - 1];
365  short side = indices[indices.length - 2];
366  CompoundTerm comp = (CompoundTerm) inh.componentAt(side);
367  if ((comp instanceof Product)) {
368  if (side == 0) {
369  subject = comp.componentAt(index);
370  predicate = ImageExt.make((Product) comp, inh.getPredicate(), index, memory);
371 
372  } else {
373  subject = ImageInt.make((Product) comp, inh.getSubject(), index, memory);
374  predicate = comp.componentAt(index);
375 
376  }
377  } else if ((comp instanceof ImageExt) && side == 1) {
378  if (index == ((ImageExt) comp).getRelationIndex()) {
379  subject = Product.make(comp, inh.getSubject(), index, memory);
380  predicate = comp.componentAt(index);
381  } else {
382  subject = comp.componentAt(index);
383  predicate = ImageExt.make((ImageExt) comp, inh.getSubject(), index, memory);
384  }
385  } else {
386  if (comp instanceof ImageInt && side == 0) if (index != ((ImageInt) comp).getRelationIndex()) {
387  subject = ImageInt.make((ImageInt) comp, inh.getPredicate(), index, memory);
388  predicate = comp.componentAt(index);
389  } else {
390  subject = comp.componentAt(index);
391  predicate = Product.make(comp, inh.getPredicate(), index, memory);
392  }
393  else {
394  return;
395  }
396  }
397  Inheritance newInh = Inheritance.make(subject, predicate, memory);
398  Term content = null;
399  if (indices.length != 2) if (!(oldContent instanceof Statement) || 1 != indices[0]) {
400  ArrayList<Term> componentList;
401  Term condition = oldContent.componentAt(0);
402  if ((oldContent instanceof Implication || (oldContent instanceof Equivalence))) {
403  if (!(condition instanceof Conjunction)) {
404  componentList = oldContent.cloneComponents();
405  componentList.set(indices[0], newInh);
406  content = Statement.make((Statement) oldContent, componentList.get(0), componentList.get(1), memory);
407  } else {
408  componentList = ((CompoundTerm) condition).cloneComponents();
409  componentList.set(indices[1], newInh);
410  Term newCond = CompoundTerm.make((CompoundTerm) condition, componentList, memory);
411  content = Statement.make((Statement) oldContent, newCond, ((Statement) oldContent).getPredicate(), memory);
412  }
413  } else {
414  componentList = oldContent.cloneComponents();
415  componentList.set(indices[0], newInh);
416  if (oldContent instanceof Conjunction) {
417  content = CompoundTerm.make(oldContent, componentList, memory);
418  }
419  }
420  } else {
421  content = Statement.make((Statement) oldContent, oldContent.componentAt(0), newInh, memory);
422  }
423  else {
424  content = newInh;
425  }
426  if (null != content) {
427  Sentence sentence = memory.getCurrentTask().getSentence();
428  TruthValue truth = sentence.getTruth();
429  BudgetValue budget;
430  budget = sentence.isQuestion() ? compoundBackward(content, memory) : compoundForward(truth, content, memory);
431  memory.singlePremiseTask(content, truth, budget);
432  }
433  }

448  private static void transformSubjectPI(CompoundTerm subject,
449  Term predicate, Memory memory) {
450  TruthValue truth = memory.getCurrentTask().getSentence().getTruth();
451  BudgetValue budget;
452  Inheritance inheritance;
453  Term newSubj, newPred;
454  if (subject instanceof Product) {
455  Product product = (Product) subject;
456  for (short i = 0; i < product.size(); i++) {
457  newSubj = product.componentAt(i);
458  newPred = ImageExt.make(product, predicate, i, memory);
459  inheritance = Inheritance.make(newSubj, newPred, memory);
460  budget = truth == null ? compoundBackward(inheritance,
461  memory) : compoundForward(truth,
462  inheritance, memory);
463  memory.singlePremiseTask(inheritance, truth, budget);
464  }
465  } else {
466  if (!(subject instanceof ImageInt)) {
467  return;
468  }
469  ImageInt image = (ImageInt) subject;
470  int relationIndex = image.getRelationIndex();
471  for (short i = 0; i < image.size(); i++) {
472  if (i == relationIndex) {
473  newSubj = image.componentAt(relationIndex);

475  memory);

478  memory);
479  newPred = image.componentAt(i);
480  }
