45  public static void reason(TaskLink tLink, TermLink bLink, Memory memory) {
46  Task task = memory.getCurrentTask();
47  Sentence taskSentence = task.getSentence();
48  Term taskTerm = (Term) taskSentence.getContent().clone(); // cloning for
49  // substitution
50  Term beliefTerm = (Term) bLink.getTarget().clone(); // cloning for
51  // substitution
52  Concept beliefConcept = memory.termToConcept(beliefTerm);
53  Sentence belief = null;
54  if (beliefConcept != null) {
55  belief = beliefConcept.getBelief(task);
56  }
57  memory.setCurrentBelief(belief); // may be null
58  if (belief != null) {
59  LocalRules.match(task, belief, memory);
60  }
61  if (!memory.noResult()) {
62  return;
63  }
64  short tIndex = tLink.getIndex(0);
65  short bIndex = bLink.getIndex(0);
66  switch (tLink.getType()) { // dispatch first by TaskLink type
67  case TermLink.SELF :
68  switch (bLink.getType()) {
69  case TermLink.COMPONENT :
70  compoundAndSelf((CompoundTerm) taskTerm, beliefTerm,
71  true, memory);
72  break;
73  case TermLink.COMPOUND :
74  compoundAndSelf((CompoundTerm) beliefTerm, taskTerm,
75  false, memory);
76  break;
77  case TermLink.COMPONENT_STATEMENT :
78  if (belief != null) {
79  SyllogisticRules.detachment(task.getSentence(),
80  belief, bIndex, memory);
81  }
82  break;
83  case TermLink.COMPOUND_STATEMENT :
84  if (belief != null) {
85  SyllogisticRules.detachment(belief,
86  task.getSentence(), bIndex, memory);
87  }
88  break;
89  case TermLink.COMPONENT_CONDITION :
90  if (belief != null) {
91  bIndex = bLink.getIndex(1);
92  SyllogisticRules.conditionalDedInd(
93  (Implication) taskTerm, bIndex, beliefTerm,
94  tIndex, memory);
95  }
96  break;
97  case TermLink.COMPOUND_CONDITION :
98  if (belief != null) {
99  bIndex = bLink.getIndex(1);
100  SyllogisticRules.conditionalDedInd(
101  (Implication) beliefTerm, bIndex, taskTerm,
102  tIndex, memory);
103  }
104  break;
105  }
106  break;
107  case TermLink.COMPOUND :
108  switch (bLink.getType()) {
109  case TermLink.COMPOUND :
110  compoundAndCompound((CompoundTerm) taskTerm,
111  (CompoundTerm) beliefTerm, memory);
112  break;
113  case TermLink.COMPOUND_STATEMENT :
114  compoundAndStatement((CompoundTerm) taskTerm, tIndex,
115  (Statement) beliefTerm, bIndex, beliefTerm,
116  memory);
117  break;
118  case TermLink.COMPOUND_CONDITION :
119  if (belief != null) {
120  if (beliefTerm instanceof Implication) {
121  SyllogisticRules.conditionalDedInd(
122  (Implication) beliefTerm, bIndex,
123  taskTerm, -1, memory);
124  } else if (beliefTerm instanceof Equivalence) {
125  SyllogisticRules.conditionalAna(
126  (Equivalence) beliefTerm, bIndex,
127  taskTerm, -1, memory);
128  }
129  }
130  break;
131  }
132  break;
133  case TermLink.COMPOUND_STATEMENT :
134  switch (bLink.getType()) {
135  case TermLink.COMPONENT :
136  componentAndStatement(
137  (CompoundTerm) memory.getCurrentTerm(), bIndex,
138  (Statement) taskTerm, tIndex, memory);
139  break;
140  case TermLink.COMPOUND :
141  compoundAndStatement((CompoundTerm) beliefTerm, bIndex,
142  (Statement) taskTerm, tIndex, beliefTerm,
143  memory);
144  break;
145  case TermLink.COMPOUND_STATEMENT :
146  if (belief != null) {
147  // bIndex = bLink.getIndex(1);
148  syllogisms(tLink, bLink, taskTerm, beliefTerm,
149  memory);
150  }
151  break;
152  case TermLink.COMPOUND_CONDITION :
153  if (belief != null) {
154  bIndex = bLink.getIndex(1);
155  if (beliefTerm instanceof Implication) {
156  conditionalDedIndWithVar(
157  (Implication) beliefTerm, bIndex,
158  (Statement) taskTerm, tIndex, memory);
159  }
160  }
161  break;
162  }
163  break;
164  case TermLink.COMPOUND_CONDITION :
165  switch (bLink.getType()) {
166  case TermLink.COMPOUND_STATEMENT :
167  if (belief != null) {
168  if (taskTerm instanceof Implication)
169  // TODO maybe put instanceof test within
170  // conditionalDedIndWithVar()
171  conditionalDedIndWithVar(
172  (Implication) taskTerm, tIndex,
173  (Statement) beliefTerm, bIndex, memory);
174  }
175  break;
176  }
177  break;
178  }

197  private static void syllogisms(TaskLink tLink, TermLink bLink,
198  Term taskTerm, Term beliefTerm, Memory memory) {
199  Sentence taskSentence = memory.getCurrentTask().getSentence();
200  Sentence belief = memory.getCurrentBelief();
201  int figure;
202  if (taskTerm instanceof Inheritance) {
203  if (beliefTerm instanceof Inheritance) {
204  figure = indexToFigure(tLink, bLink);
205  asymmetricAsymmetric(taskSentence, belief, figure, memory);
206  } else if (beliefTerm instanceof Similarity) {
207  figure = indexToFigure(tLink, bLink);
208  asymmetricSymmetric(taskSentence, belief, figure, memory);
209  } else {
210  detachmentWithVar(belief, taskSentence, bLink.getIndex(0),
211  memory);
212  }
213  } else if (taskTerm instanceof Similarity) {
214  if (beliefTerm instanceof Inheritance) {
215  figure = indexToFigure(bLink, tLink);
216  asymmetricSymmetric(belief, taskSentence, figure, memory);
217  } else if (beliefTerm instanceof Similarity) {
218  figure = indexToFigure(bLink, tLink);
219  symmetricSymmetric(belief, taskSentence, figure, memory);
220  }
221  } else if (taskTerm instanceof Implication) {
222  if (beliefTerm instanceof Implication) {
223  figure = indexToFigure(tLink, bLink);
224  asymmetricAsymmetric(taskSentence, belief, figure, memory);
225  } else if (beliefTerm instanceof Equivalence) {
226  figure = indexToFigure(tLink, bLink);
227  asymmetricSymmetric(taskSentence, belief, figure, memory);
228  } else if (beliefTerm instanceof Inheritance) {
229  detachmentWithVar(taskSentence, belief, tLink.getIndex(0),
230  memory);
231  }
232  } else if (taskTerm instanceof Equivalence) {
233  if (beliefTerm instanceof Implication) {
234  figure = indexToFigure(bLink, tLink);
235  asymmetricSymmetric(belief, taskSentence, figure, memory);
236  } else if (beliefTerm instanceof Equivalence) {
237  figure = indexToFigure(bLink, tLink);
238  symmetricSymmetric(belief, taskSentence, figure, memory);
239  } else if (beliefTerm instanceof Inheritance) {
240  detachmentWithVar(taskSentence, belief, tLink.getIndex(0),
241  memory);
242  }
243  }

272  private static void asymmetricAsymmetric(Sentence sentence,
273  Sentence belief, int figure, Memory memory) {
274  Statement s1 = (Statement) sentence.cloneContent();
275  Statement s2 = (Statement) belief.cloneContent();
276  Term t1, t2;
277  switch (figure) {
278  case 11 : // induction
279  if (Variable.unify(Symbols.VAR_INDEPENDENT, s1.getSubject(),
280  s2.getSubject(), s1, s2)) {
281  if (s1.equals(s2)) {
282  return;
283  }
284  t1 = s2.getPredicate();
285  t2 = s1.getPredicate();
286  SyllogisticRules.abdIndCom(t1, t2, sentence, belief,
287  figure, memory);
288  CompositionalRules.composeCompound(s1, s2, 0, memory);
289  }
290 
291  break;
292  case 12 : // deduction
293  if (Variable.unify(Symbols.VAR_INDEPENDENT, s1.getSubject(),
294  s2.getPredicate(), s1, s2)) {
295  if (s1.equals(s2)) {
296  return;
297  }
298  t1 = s2.getSubject();
299  t2 = s1.getPredicate();
300  if (Variable.unify(Symbols.VAR_QUERY, t1, t2, s1, s2)) {
301  LocalRules.matchReverse(memory);
302  } else {
303  SyllogisticRules.dedExe(t1, t2, sentence, belief,
304  memory);
305  }
306  }
307  break;
308  case 21 : // exemplification
309  if (Variable.unify(Symbols.VAR_INDEPENDENT, s1.getPredicate(),
310  s2.getSubject(), s1, s2)) {
311  if (s1.equals(s2)) {
312  return;
313  }
314  t1 = s1.getSubject();
315  t2 = s2.getPredicate();
316  if (Variable.unify(Symbols.VAR_QUERY, t1, t2, s1, s2)) {
317  LocalRules.matchReverse(memory);
318  } else {
319  SyllogisticRules.dedExe(t1, t2, sentence, belief,
320  memory);
321  }
322  }
323  break;
324  case 22 : // abduction
325  if (Variable.unify(Symbols.VAR_INDEPENDENT, s1.getPredicate(),
326  s2.getPredicate(), s1, s2)) {
327  if (s1.equals(s2)) {
328  return;
329  }
330  t1 = s1.getSubject();
331  t2 = s2.getSubject();
332  if (!SyllogisticRules
333  .conditionalAbd(t1, t2, s1, s2, memory)) { // if
334  // conditional
335  // abduction,
336  // skip
337  // the
338  // following
339  SyllogisticRules.abdIndCom(t1, t2, sentence, belief,
340  figure, memory);
341  CompositionalRules.composeCompound(s1, s2, 1, memory);
342  }
343  }
344  break;
345  default :
346  }
